
% Billede tekst del
\newenvironment{BilledeTekst}[4]{ 
    \begin{figure}[H]
        \centering
        %\def\svgwidth{\columnwidth}
        \includegraphics[width=#4,height=\textheight,keepaspectratio]{#1}
        \caption{\textit{#2}}
        \label{fig:#3}
    \end{figure}
}



% Side by side Billede tekst del
\newenvironment{SBSBilledeTekst}[8]{
    \begin{figure}[H]
        \centering
      \begin{subfigure}[b]{0.2\textwidth}
        \includegraphics[width=\textwidth]{#1}
        \caption{\textit{#2}}
        \label{fig:#3}
      \end{subfigure}
      %
      \begin{subfigure}[b]{0.2\textwidth}
        \includegraphics[width=\textwidth]{#4}
        \caption{\textit{#5}}
        \label{fig:#6}
      \end{subfigure}
    \caption{\textit{#7}}
    \label{fig:#8}
    \end{figure}
}
